//Задание 1 (функции)
#include <stdio.h>

long long factorial(int n) {
    if (n < 0) return -1;
    long long fact = 1;
    for (int i = 1; i <= n; i++) {
        fact *= i;
    }
    return fact;
}

int main() {
    int num;
    printf("Введите число: ");
    scanf("%d", &num);

    long long result = factorial(num);
    if (result == -1) {
        printf("Факториал не определен для отрицательных чисел.\n");
    } else {
        printf("Факториал %d = %lld\n", num, result);
    }
    return 0;
}

//Задание 2 (функции)
#include <stdio.h>
#include <math.h>

int isPrime(int n) {
    if (n < 2) return 0;
    if (n == 2) return 1;
    if (n % 2 == 0) return 0;

    for (int i = 3; i <= sqrt(n); i += 2) {
        if (n % i == 0) return 0;
    }
    return 1;
}

int main() {
    int num;
    printf("Введите число: ");
    scanf("%d", &num);

    if (isPrime(num)) {
        printf("%d — простое число.\n", num);
    } else {
        printf("%d — не простое число.\n", num);
    }
    return 0;
}

void printDivisors(int n) {
    printf("Делители числа %d: ", n);
    for (int i = 1; i <= n; i++) {

        }

}

    scanf("%d", &num);
    printDivisors(num);
    return 0;

//Задание 4 (функции)
#include <stdio.h>

void printNumbersBetween(int a, int b) {
    if (a > b) {
        int temp = a;
        a = b;
        b = temp;
    }

    printf("Числа между %d и %d: ", a, b);
    for (int i = a + 1; i < b; i++) {
        printf("%d ", i);
    }
    printf("\n");
}

int main() {
    int x, y;
    printf("Введите два числа: ");
    scanf("%d %d", &x, &y);
    printNumbersBetween(x, y);
    return 0;
}

//Задание 5 (функции)
#include <stdio.h>

int sumOfDivisors(int n) {
    int sum = 0;
    for (int i = 1; i < n; i++) {
        if (n % i == 0) {
            sum += i;
        }
    }
    return sum;
}

int areAmicable(int a, int b) {
    return (sumOfDivisors(a) == b && sumOfDivisors(b) == a);
}

int main() {
    int x, y;
    printf("Введите два числа: ");
    scanf("%d %d", &x, &y);

    if (areAmicable(x, y)) {
        printf("%d и %d — дружественные числа.\n", x, y);
    } else {
        printf("%d и %d — не дружественные числа.\n", x, y);
    }
    return 0;
}
